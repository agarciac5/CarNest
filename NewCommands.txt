# Levantar app + PostgreSQL (migraciones al arrancar el contenedor web)
docker compose up --build

# Migraciones manuales dentro del contenedor
docker compose exec web python manage.py makemigrations
docker compose exec web python manage.py migrate

# Datos de ejemplo (opcional)
docker compose exec web python manage.py seed_inventario

# Superusuario david (desarrollo)

python manage.py create_superuser_david
